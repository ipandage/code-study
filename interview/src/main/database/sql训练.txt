统计每个地区的合同金额合计并按此倒序排列显示
select region, sum(total) totals from test group by region order by totals desc

统计出每个地区的销售人员数量
select region, count(distinct sales) totals from test group by region

统计每个地区合同金额最少的销售人员
SELECT
region,
sales,
MIN(total)
FROM test a WHERE region IN (SELECT DISTINCT (region) FROM test b WHERE a.region=b.region)
GROUP BY region ;

统计出所有超过本地区合同金额平均值的合同（orderId为合同id外键）及金额
SELECT
orderId,
region,
total
FROM
test a
WHERE total >=(SELECT AVG(total) FROM test b WHERE a.region=b.region GROUP BY region )
GROUP BY region ;
